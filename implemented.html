def get_data(offline=False):
    if offline:
        df=cf.datagen.ohlc()
        df=df.rename(columns=dict([(_,_.title()) for _ in df.columns]))
    else:
        df=pdr.get_data_yahoo('GS','2008-01-01','2009-01-01')
    return df
